#include <iostream>
#include <windows.h>
#include <math.h>

using namespace std;
void main()
{
	SetConsoleOutputCP(1251);
	int m = 100, i = 0, p=0, o=0;
	double* x = new double[m];
	x[i] = 0;
	i = 2;
	x[0] = 0.3;
	x[1] = -0.3;
	while (i < m)
	{
		x[i] = i + sin(x[i - 2]);
		i++;
	}
	
	cout << "Получим следуюший одномерный массив \n";
	i = 0;
	do
	{
		cout << x[i] << "\t";
		i++;
	} while (i < m);
	i = 0;

	for (int i = 0; i < m; i++) 
	{
		if (x[i] > 0) 
		{
			p = p + 1;
			
			}
		else 
		{
			o = o + 1;
		}
		
	}
	cout << "\nКоличество положительных чисел:"<< p<< ""<< "\nКоличество отрицательных чисел:"<< o;
	cout << endl;
	system("pause");
}
